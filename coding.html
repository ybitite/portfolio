    }

    // DAO (Data Access Object) pour accéder aux données
    @Dao
    public interface UserDao {
      @Insert
      void insert(User user);

      @Query("SELECT * FROM users WHERE id = :userId")
      User getUserById(int userId);

      @Delete
      void delete(User user);
    }

    // Base de données Room
    @Database(entities = {User.class}, version = 1)
    public abstract class AppDatabase extends RoomDatabase {
      public abstract UserDao userDao();

Implémentation d'un RecyclerView avec un Adaptateur Personnalisé
Affiche une liste d'éléments en utilisant RecyclerView avec un adaptateur personnalisé pour une meilleure performance et flexibilité.

Android
Java
RecyclerView
UI/UX



    // Classe de l'adaptateur pour RecyclerView
    public class MyAdapter extends RecyclerView.Adapter {
      private List data;

      // Constructeur
      public MyAdapter(List data) {
          this.data = data;
      }

      // Création des nouvelles vues
      @Override
      public ViewHolder onCreateViewHolder(ViewGroup parent, int viewType) {
          // Crée une nouvelle vue
          View view = LayoutInflater.from(parent.getContext())
                  .inflate(R.layout.recycler_item, parent, false);
          return new ViewHolder(view);
      }

      // Remplacement du contenu d'une vue
      @Override
      public void onBindViewHolder(ViewHolder holder, int position) {
          // Récupère l'élément pour cette position
          String item = data.get(position);
          // Remplace le contenu de la vue
          holder.textView.setText(item);
      }

      // Retourne la taille de l'ensemble de données
      @Override
      public int getItemCount() {
          return data.size();
      }

      // Classe ViewHolder
      public static class ViewHolder extends RecyclerView.ViewHolder {
          public TextView textView;

          public ViewHolder(View view) {
              super(view);
              textView = view.findViewById(R.id.textView);
          }
      }
    }
